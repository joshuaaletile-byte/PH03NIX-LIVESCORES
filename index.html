// Fetch real live matches from backend async function renderMatches() { const container = document.getElementById("matches"); container.innerHTML = ""; try { const res = await fetch('https://https://ph03nix-backend.onrender.com/api/live-matches'); const matches = await res.json(); matches.forEach(match => { container.innerHTML += `
${match.league} LIVE
${match.home} ${match.score} ${match.away}
${match.minute}
`; }); } catch (err) { container.innerHTML = '

Failed to load live matches

'; } }
